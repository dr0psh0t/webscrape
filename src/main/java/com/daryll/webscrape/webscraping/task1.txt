method scrapeContactForm(Client client, List companies)

    initialize empty list of filledDocuments

    loop through companies list
        scrape homepage
        look for a text "contact us" or "contact" (ignore case)
        if "contact us" or "contact" text is found
            if "contact us" or "contact" is a hyperlink
                scrape that "contact us" hyperlink
                    look for a "submit" button
                    if "submit" button is found
                        newDoc = scrapeContactForm(formElement, client)
                        store newDoc to filledDocuments
            else if "contact us" or "contact" is just a text
                look for a "submit" button
                if "submit" button is found
                    newDoc = scrapeContactForm(formElement, client)
                    store newDoc to filledDocuments


    return filledDocuments
